CREATE DATABASE IF NOT EXISTS ceti_sensordata;
USE ceti_sensordata;

CREATE TABLE IF NOT EXISTS sensores (
  id INT NOT NULL AUTO_INCREMENT,
  nombre VARCHAR(255) NOT NULL,
  PRIMARY KEY (id)
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS datos (
  id INT NOT NULL AUTO_INCREMENT,
  sensor_id INT NOT NULL,
  tipoDeDato VARCHAR(255) NOT NULL,
  resultado FLOAT NOT NULL,
  PRIMARY KEY (id),
  CONSTRAINT fk_sensor_id FOREIGN KEY (sensor_id) REFERENCES sensores(id) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB;
